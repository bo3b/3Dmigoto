  o2.xyzw = cxadd.xyzw;

  o4.y = dot(v2.xyzw, mvp[1].xyzw);

float4 params = IniParams.Load(0);
float depth = world_z_from_depth_buffer(0,0);

float aspectRatio = resolution.x / resolution.y;
float cutLeft = -0.45;
float cutRight = 0.3;
float cutUp = 0.0;
float cutDown = -0.0;

if (params.z >= 1.0)
{
	o4.x -= stereo.x* ( o4.z - stereo.y);
}
else
{
	o4.x += stereo.x* params.x;
}

  return;
}
