{
  float4 r0;
  uint4 bitmask, uiDest;
  float4 fDest;

  o0.xyzw = v0.xyzw;
  o1.xyzw = v1.xyzw;
  r0.xy = v1.zz * float2(5.100200e+002,5.100200e+002) + float2(1.000000e-001,1.100000e+000);
  r0.xy = (uint2)r0.xy;
  o2.x = dot(v2.xyzw, vfuniforms[r0.x].xyzw);
  o2.y = dot(v2.xyzw, vfuniforms[r0.y].xyzw);
  o2.zw = float2(0.000000e+000,1.000000e+000);

if (!rt_is_screen())
	return;

float4 stereo = StereoParams.Load(0);
float4 params = IniParams.Load(0);
if (params.z == 1)
	o2.x += adjust_from_stereo2mono_depth_buffer(dot(float4(0,0,0,1), vfuniforms[r0.x].xyzw), dot(float4(0,0,0,1), vfuniforms[r0.y].xyzw));
else if (params.z == 2)
	o2.x += adjust_from_stereo2mono_depth_buffer(o2.x, o2.y);
else
	o2.x -= stereo.x*(params.x);  
  return;
}
