  float4 fDest;

  r0.x = (uint)v2.x << 1;
  r0.x = r0.x;
  r0.xy = float2(1.000000e-001,1.100000e+000) + r0.xx;
  r0.xy = (uint2)r0.xy;
  o1.x = dot(v1.xyzw, vfuniforms[r0.x].xyzw);

{
	o1.x -= stereo.x* ( o1.z - stereo.y);
}
else
{
	o1.x += stereo.x* params.x;
} 
	
  return;
}
